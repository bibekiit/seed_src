0014 function [cimg] = compute_coherence(oimg)
0015     [h,w]   =   size(oimg);
0016     cimg    =   zeros(h,w);
0017     N       =   2;
0018     %---------------
0019     %pad the image
0020     %---------------
0021     oimg    =   [flipud(oimg(1:N,:));oimg;flipud(oimg(h-N+1:h,:))]; %pad the rows
0022     oimg    =   [fliplr(oimg(:,1:N)),oimg,fliplr(oimg(:,w-N+1:w))]; %pad the cols
0023     %compute coherence
0024     for i=N+1:h+N
0025         for j = N+1:w+N
0026             th  = oimg(i,j);
0027             blk = oimg(i-N:i+N,j-N:j+N);
0028             cimg(i-N,j-N)=sum(sum(abs(cos(blk-th))))/((2*N+1).^2);
0029         end;
0030     end;
